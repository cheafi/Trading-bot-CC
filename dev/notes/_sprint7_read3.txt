 303:     def __init__(self):

 307:     def load_calibration(self, rows: List[Dict]):
 308:         """Load pre-computed regime edge stats from DB."""
 309:         for r in rows:
 310:             key = r.get("calibration_bucket", "")
 311:             self._calibration_cache[key] = r

 317:         features: Dict[str, Any],
